Salmon Data Summary

annual_esc <- esc %>%
  separate(sampleDate, into = c("year", "month", "day"), sep = "-") %>%
  mutate(year = as.numeric(year) )%>%
  group_by(year, Species, SASAP.Region)%>%
  summarise(totalCount = sum(DailyCount))%>%
  filter(Species %in% c("Sockeye", "Chum", "Coho", "Chinook", "Pink"))
